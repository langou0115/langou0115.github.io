模板

将强连通分量缩成一个点，变成有向无环图（DAG）

template


int degree[MAXN]; // 有向无环图每个点的入度
int tb[MAXN], tb_a[MAXN], cnt = 1, times = 1;
// tb记录原来的点映射到新的点是什么， tb_a记录新的点所花费的时间, cnt记录新点的下标
void tarjan(int x)
{
	dfn[x] = low[x] = times++;
	instack[x] = 1;
	s.push(x);
	for(int i = 0; i < G[x].size(); i++){
		int y = G[x][i];
		if(!dfn[y]){
			tarjan(y);
			low[x] = min(low[x], low[y]);
		}else if(instack[y]){
			low[x] = min(low[x], dfn[y]);
		}
	}
	 // 重点在此
	if(dfn[x] == low[x]){
		int tmp;
		do{
			tmp = s.top();
			s.pop();
			instack[tmp] = 0;
			tb[tmp] = cnt;
			tb_a[cnt] += a[tmp];
		}while(tmp != x);
		dp[cnt] = tb_a[cnt];
		cnt++;
	}
}

void build_tb()
{
	for(int i = 0; i < n; i++){
		for(int j = 0; j < G[i].size(); j++){
			int y = G[i][j];
			if(tb[i] != tb[y]){ // 映射点相同不做处理
				TG[tb[i]].push_back(tb[y]);
				indegree[tb[y]]++;
			}
		}
	}
}
